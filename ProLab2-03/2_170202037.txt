    private void button_girisActionPerformed(java.awt.event.ActionEvent evt) {                                             
        
        PreparedStatement st;
        ResultSet rs;
        
        String kullaniciemail = jText_Mail.getText();
        String kullanicisifre = String.valueOf(jPass.getPassword());
        
        try{
        
        String sql = "select * from 'kullanici'";
        String query = "select * from 'kullanici' where 'kullaniciemail' = ? AND 'kullanicisifre' = ?";
            
            st = Proje203.getConnection().prepareStatement(query);
            st.setString(1, "kullaniciemail");
            st.setString(2, "kullanicisifre");
            rs = st.executeQuery();
            if(rs.next()){
                AnaMenu menu = new AnaMenu();
                menu.setVisible(true);
                menu.pack();
                menu.setLocationRelativeTo(null);
                // close the current form(Giris form)
                this.dispose();
            }
            else{
                // error message
                JOptionPane.showMessageDialog(null, "E-Mail veya Şifre Hatalı!","Giriş Hatası",2);
            }

        }catch(SQLException e){
            System.out.println(e.getMessage());
        }
        
        
        
    }

    public boolean verifyFields()
    {
         String kullaniciadi = jText_username.getText();
         String kullaniciemail =  jText_email.getText();
         String kullanicisifre = String.valueOf(jPass.getPassword());
         String kullanicidogumtarihi = jText_birthday.getText();
        
        
        // check empty fields
        if(kullaniciadi.trim().equals("") || kullaniciemail.trim().equals("") || kullanicisifre.trim().equals("")
           || kullanicidogumtarihi.trim().equals("") )
        {
            JOptionPane.showMessageDialog(null, "Bir veya daha fazla alan boş.","Boş Alanlar",2);
            return false;
        }else{
            return true;
        }
    }
    
    public boolean checkkullaniciadi(String kullaniciadi){
        
        PreparedStatement st;
        ResultSet rs;
        boolean kullaniciadi_exist = false;
        
        String query = "SELECT * FROM `kullanici` WHERE `kullaniciadi` = ?";
        
        try {
            
            st = Proje203.getConnection().prepareStatement(query);
            st.setString(1, kullaniciadi);
            rs = st.executeQuery();
        
            if(rs.next())
            {
                kullaniciadi_exist = true;
                JOptionPane.showMessageDialog(null, "Bu kullanıcı adı alınmış, başka bir tane deneyin", "Kullanıcı adı hatası", 2);
            }
            
        } catch (SQLException ex) {
            Logger.getLogger(Giris.class.getName()).log(Level.SEVERE, null, ex);
        }
        
        return kullaniciadi_exist;
    }
    
    public boolean checkEmail(String kullaniciemail){
